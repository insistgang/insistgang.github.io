void Merge(LinkList A,LinkList B,LinkList &C){
    LNode *p=A->next;
    LNode *q=B->next;
    C=A;
    C->next=NULL;
    LNode *r=C;
    while(p!=NULL&&q!=NULL){
        if(p->data<q->data){
            r->next=p;
            p=p->next;
            r=r->next;
        }else{
            r->next=q;
            q=q->next;
            r=r->next;
        }
    }
