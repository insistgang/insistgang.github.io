 用冒泡法对一组用户输入的整数进行排序。 

```cpp
#include<iostream>
#include<cstdio>
using namespace std;
/*
外层n-1次循环
每次循环都在待排序的序列中把最大的移到待排序的最后面1~n-i-1

*/
int main(){
    int a[10]={3,2,5,1,4,8,9,7,10,6};
    for(int i=0;i<9;i++){
        for(int j=1;j<=10-i-1;j++){
            if(a[j]>a[j-1])
            swap(a[j-1],a[j]);
        }
    }
    for(int i=0;i<10;i++)cout<<a[i]<<" ";
    return 0;
}
```

```cpp
    int len=strlen(a);

    while(i<len){
        if(a[i]>='a'&&a[i]<='z'||a[i]>='A'&&a[i]<='Z'){
            temp=0;
            while(a[i]>='a'&&a[i]<='z'||a[i]>='A'&&a[i]<='Z'){
                i++;
                temp++;
            }
            if(temp>length){
                length=temp;
                pos=i-1;
            }
        }else{
            i++;
        }
    }

        cout<<a[i];
```
